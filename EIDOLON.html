import cv2
import numpy as np
import soundfile as sf
import math

def generate_tone(freq, duration, sr=44100, wave_type='sine'):
    t = np.linspace(0, duration, int(sr * duration), False)
    if wave_type == 'sine': return np.sin(freq * t * 2 * np.pi)
    elif wave_type == 'sawtooth': return 2 * (t * freq - np.floor(0.5 + t * freq))
    return np.zeros_like(t)

def apply_envelope(audio, sr, attack=0.1, decay=0.1):
    attack_samples = int(attack * sr)
    decay_samples = int(decay * sr)
    env = np.ones_like(audio)
    if len(audio) > attack_samples + decay_samples:
        env[:attack_samples] = np.linspace(0, 1, attack_samples)
        env[-decay_samples:] = np.linspace(1, 0, decay_samples)
    return audio * env

def generate_eidolon_audio(image_path, output_path='voynich_output.wav'):
    img = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)
    img_blurred = cv2.GaussianBlur(img, (5, 5), 0)
    edges = cv2.Canny(img_blurred, 50, 150)

    contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    lines = cv2.HoughLinesP(edges, 1, np.pi/180, threshold=50, minLineLength=30, maxLineGap=10)

    sr = 44100
    duration = 12.0
    audio_mix = np.zeros(int(sr * duration))
    root_freq = 41.20 
    theta_lock = 6.55 

    if lines is not None:
        for i in range(min(len(lines), 8)):
            x1, y1, x2, y2 = lines[i][0]
            length = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
            drone_freq = root_freq + (max(1, int(length / 50)) * theta_lock)
            tone = apply_envelope(generate_tone(drone_freq, duration, sr, 'sawtooth'), sr, 2.0, 3.0)
            audio_mix += tone * 0.15 

    if contours:
        valid_contours = [c for c in contours if cv2.contourArea(c) > 50]
        for i in range(min(len(valid_contours), 16)):
            area = cv2.contourArea(valid_contours[i])
            vertices = len(cv2.approxPolyDP(valid_contours[i], 0.04 * cv2.arcLength(valid_contours[i], True), True))
            note_freq = (root_freq * 4) + (vertices * theta_lock * 2) 
            note_duration = 0.2 + (area / 10000.0) 
            tone = apply_envelope(generate_tone(note_freq, note_duration, sr, 'sine'), sr, 0.05, 0.1)
            
            x_pos = cv2.boundingRect(valid_contours[i])[0]
            start_idx = int(((x_pos / img.shape[1]) * (duration - note_duration)) * sr)
            audio_mix[start_idx:start_idx + len(tone)] += tone * 0.25 

    max_amp = np.max(np.abs(audio_mix))
    if max_amp > 0: audio_mix = audio_mix / max_amp
    sf.write(output_path, audio_mix, sr)

if __name__ == "__main__":
    generate_eidolon_audio('manuscript.jpg', 'eidolon_voynich_01.wav')
