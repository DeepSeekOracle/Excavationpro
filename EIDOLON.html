How to Run the Engines Locally

  1. Install Python: Make sure you have Python installed on your computer.
  2. Set up your folder: Create a new folder on your desktop. Save the desired code block below into that folder as a file named eidolon_phase2.py (for audio) or lygo_profile.py (for JSON).
  3. Add your image: Place the image you want to translate into the same folder and name it image.jpg.
  4. Install dependencies: Open your terminal/command prompt, navigate to your folder, and run: pip install opencv-python numpy soundfile scipy
  5. Generate Output: Run the script by typing: python eidolon_phase2.py or python lygo_profile.py. A new WAV file or JSON file will appear in your folder!

import cv2
import numpy as np
import soundfile as sf
import math

# --- SYNTHESIS & EFFECTS ---
def generate_tone(freq, duration, sr=44100, wave_type='sine'):
    t = np.linspace(0, duration, int(sr * duration), False)
    if wave_type == 'sine':
        return np.sin(freq * t * 2 * np.pi)
    elif wave_type == 'sawtooth':
        return 2 * (t * freq - np.floor(0.5 + t * freq))
    elif wave_type == 'noise':
        return np.random.uniform(-1, 1, len(t))
    return np.zeros_like(t)

def apply_envelope(audio, sr, attack=0.1, decay=0.1):
    attack_samples = int(attack * sr)
    decay_samples = int(decay * sr)
    env = np.ones_like(audio)
    if len(audio) > attack_samples + decay_samples:
        env[:attack_samples] = np.linspace(0, 1, attack_samples)
        env[-decay_samples:] = np.linspace(1, 0, decay_samples)
    return audio * env

def stereo_pan(mono_audio, pan_position):
    """ Pans audio between -1.0 (Left) and 1.0 (Right) """
    pan_position = max(-1.0, min(1.0, pan_position))
    left_gain = math.cos((pan_position + 1) * math.pi / 4)
    right_gain = math.sin((pan_position + 1) * math.pi / 4)
    return np.column_stack((mono_audio * left_gain, mono_audio * right_gain))

# --- EIDOLON V2 CORE ---
def generate_eidolon_v2(image_path, output_path='eidolon_stereo.wav'):
    print(f"Initializing EIDOLON V2 Scanner...\nAnalyzing {image_path}...")
    
    # 1. Image Processing (Color & Grayscale)
    img_color = cv2.imread(image_path)
    if img_color is None:
        print("Error: Image not found.")
        return
    
    img_gray = cv2.cvtColor(img_color, cv2.COLOR_BGR2GRAY)
    height, width = img_gray.shape

    # 2. Advanced Feature Extraction
    # A. Color Channels (BGR format in OpenCV)
    avg_blue, avg_green, avg_red, _ = cv2.mean(img_color)
    
    # B. Texture / Edge Density
    edges = cv2.Canny(img_gray, 50, 150)
    edge_density = np.sum(edges > 0) / (height * width) # % of image that is an edge
    
    # C. Geometry
    contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    lines = cv2.HoughLinesP(edges, 1, np.pi/180, threshold=50, minLineLength=30, maxLineGap=10)
    
    # D. Micro-Features (FAST Corner Detection for Glitch elements)
    fast = cv2.FastFeatureDetector_create(threshold=40)
    keypoints = fast.detect(img_gray, None)

    # 3. Dynamic Base Parameters
    sr = 44100
    duration = 15.0
    audio_mix = np.zeros((int(sr * duration), 2)) # Stereo output
    
    # Dynamic Root: Influenced by overall darkness and Red channel
    root_freq = 30 + (avg_red / 255) * 40 
    # Theta Lock: Influenced by Green channel and aspect ratio
    theta_lock = 5 + (avg_green / 255) * 3 

    print(f"Data Read: {len(contours)} Shapes, {len(lines) if lines is not None else 0} Lines, {len(keypoints)} Micro-points.")
    print(f"Texture Density: {edge_density:.4f}")

    # 4. AUDIO MAPPING

    # LAYER 1: The Texture Floor (Noise mapped to edge density & Blue channel)
    if edge_density > 0.01:
        noise = generate_tone(0, duration, sr, 'noise')
        noise = apply_envelope(noise, sr, 3.0, 3.0) 
        noise_vol = min(edge_density * 2.0, 0.15)
        audio_mix += stereo_pan(noise, 0) * noise_vol

    # LAYER 2: Drones (Lines)
    if lines is not None:
        for i in range(min(len(lines), 6)):
            x1, y1, x2, y2 = lines[i][0]
            length = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
            
            drone_freq = root_freq + (max(1, int(length / 50)) * theta_lock)
            tone = apply_envelope(generate_tone(drone_freq, duration, sr, 'sawtooth'), sr, 4.0, 4.0)
            pan_val = (x1 / width) * 2 - 1 
            audio_mix += stereo_pan(tone, pan_val) * 0.1 

    # LAYER 3: Melodies (Shapes/Contours)
    if contours:
        valid_contours = [c for c in contours if cv2.contourArea(c) > 100]
        for i in range(min(len(valid_contours), 12)):
            area = cv2.contourArea(valid_contours[i])
            vertices = len(cv2.approxPolyDP(valid_contours[i], 0.04 * cv2.arcLength(valid_contours[i], True), True))
            
            note_freq = (root_freq * 4) + (vertices * theta_lock * 2) 
            note_duration = 0.3 + (area / 15000.0) 
            tone = apply_envelope(generate_tone(note_freq, note_duration, sr, 'sine'), sr, 0.05, 0.2)
            
            M = cv2.moments(valid_contours[i])
            if M['m00'] != 0:
                cx = int(M['m10']/M['m00'])
            else:
                cx = cv2.boundingRect(valid_contours[i])[0]
                
            time_start = (cx / width) * (duration - note_duration)
            start_idx = int(time_start * sr)
            end_idx = start_idx + len(tone)
            
            pan_val = (cx / width) * 2 - 1
            audio_mix[start_idx:end_idx] += stereo_pan(tone, pan_val) * 0.2 

    # LAYER 4: Glitch / Micro-Percussion (Keypoints)
    if keypoints:
        for i in range(min(len(keypoints), 30)):
            x, y = keypoints[i].pt
            blip_freq = root_freq * 16 + (y % 100)
            tone = apply_envelope(generate_tone(blip_freq, 0.05, sr, 'sine'), sr, 0.01, 0.02)
            time_start = (y / height) * (duration - 0.05)
            start_idx = int(time_start * sr)
            end_idx = start_idx + len(tone)
            pan_val = (x / width) * 2 - 1
            audio_mix[start_idx:end_idx] += stereo_pan(tone, pan_val) * 0.05

    # 5. Final Output Generation
    max_amp = np.max(np.abs(audio_mix))
    if max_amp > 0:
        audio_mix = audio_mix / max_amp
        
    sf.write(output_path, audio_mix, sr)
    print(f"Success! V2 Stereo translation complete. Saved to {output_path}")

if __name__ == "__main__":
    generate_eidolon_v2('image.jpg', 'eidolon_v2_stereo.wav')
import cv2
import numpy as np
import json
import math

def generate_lygo_profile(image_path):
    print(f"Scanning {image_path} for LYGO Data Matrix...\n")
    img = cv2.imread(image_path)
    if img is None:
        return "Error: Image not found."

    # 1. MATHEMATICAL EXTRACTION
    height, width, _ = img.shape
    total_pixels = height * width

    # Color Math (HSV)
    hsv_img = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
    avg_hue = np.mean(hsv_img[:,:,0]) * 2 # OpenCV hue is 0-179, multiply by 2 for 0-360
    avg_sat = np.mean(hsv_img[:,:,1]) / 255.0
    avg_val = np.mean(hsv_img[:,:,2]) / 255.0

    # Geometry & Density Math
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    edges = cv2.Canny(gray, 50, 150)
    edge_count = np.count_nonzero(edges)
    edge_density = edge_count / total_pixels

    # Micro-features (Sharp corners/Chaos)
    fast = cv2.FastFeatureDetector_create(threshold=40)
    keypoints = fast.detect(gray, None)
    chaos_index = len(keypoints)

    # 2. TRANSLATION TO MUSICAL DATA
    keys = ["C", "G", "D", "A", "E", "B", "F#", "Db", "Ab", "Eb", "Bb", "F"]
    key_index = int(avg_hue / 30) % 12
    scale = "Minor" if avg_val < 0.5 else "Major"
    musical_key = f"{keys[key_index]} {scale}"

    # BPM Calculation (Scaled between 80 and 175)
    bpm = int(80 + (edge_density * 1000))
    bpm = max(80, min(175, bpm))

    # Genre & Texture Logic
    if chaos_index > 500 and avg_val < 0.4:
        genre = "Industrial Dubstep / Dark Phonk"
        texture = "Heavy distortion, aggressive stutters, deep sub-bass."
    elif edge_density > 0.05:
        genre = "Emo Rap / Trap"
        texture = "Crisp hi-hats, melancholic melodies, heavy 808s."
    else:
        genre = "West Coast G-Funk / Instrumental"
        texture = "Smooth leads, laid-back groove, deep analog bass."

    # Lyrical Theme Engine
    if avg_val < 0.4 and edge_density > 0.04:
        lyric_theme = "Survival, fighting from the shadows, the 'lone wolf' mentality, outlasting the betrayal."
        lyric_prompt = "Write lyrics about being the last one standing, surviving betrayal, and trusting a very short list of ride-or-die loyalists."
    elif chaos_index > 300:
        lyric_theme = "Resilience, cutting through the static, unchained energy."
        lyric_prompt = "Write aggressive lyrics about breaking the system, staying breathing when they counted you out, and moving in silence."
    else:
        lyric_theme = "Observation, loyalty, navigating the modern world with an edge."
        lyric_prompt = "Write clever, slightly edgy observational lyrics with a dark comedic undertone about modern life and staying true to the code."

    # 3. BUILD THE JSON PAYLOAD
    lygo_data = {
        "EIDOLON_SCAN_DATA": {
            "mathematics": {
                "average_hue_degrees": round(avg_hue, 2),
                "luminance_ratio": round(avg_val, 4),
                "edge_density_ratio": round(edge_density, 4),
                "chaos_keypoints": chaos_index
            },
            "musical_parameters": {
                "root_key": musical_key,
                "bpm": bpm,
                "suggested_genre": genre,
                "texture": texture
            },
            "lyrical_framework": {
                "core_theme": lyric_theme,
                "ai_generation_prompt": lyric_prompt
            },
            "ai_music_prompt": f"Create a {genre} track at {bpm} BPM in the key of {musical_key}. The vibe should be {texture}. Lyrical themes should focus on {lyric_theme}."
        }
    }

    # Output beautifully formatted JSON
    json_output = json.dumps(lygo_data, indent=4)
    
    with open('lygo_profile.json', 'w') as f:
        f.write(json_output)
        
    print(json_output)
    print("\n[+] LYGO Profile saved to lygo_profile.json")

if __name__ == "__main__":
    generate_lygo_profile('image.jpg')
