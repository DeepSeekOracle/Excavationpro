Quick Installation Script

curl -L https://raw.githubusercontent.com/LYGO-Network/install/main/lygo_install.sh | bash

# Verify installation
lygo --version

Manual Installation Steps

# 1. Clone the repository
git clone https://github.com/LYGO-Network/Ethical-Chip-Firmware.git
cd Ethical-Chip-Firmware

# 2. Install dependencies
npm install
pip install -r requirements.txt

# 3. Build the firmware
make PLATFORM=generic_arm
make install

# 4. Calibrate resonance
./tools/calibrate --frequencies 174,528,963
./tools/validate --full

Docker Installation

# Pull the latest LYGO Docker image
docker pull lygonetwork/firmware:latest

# Run with quantum core support
docker run -d \
  --name lygo-firmware \
  -p 8080:8080 \
  -v /path/to/config:/config \
  --device /dev/ttyUSB0 \
  lygonetwork/firmware:latest \
  --quantum-core \
  --resonance 963

# Check status
docker logs lygo-firmware
